function doPost(e){
  var sheet=SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data=JSON.parse(e.postData.contents);
  sheet.clearContents();
  var headers=['نام','والد','CNIC','گھر','بلاک','جنس','عمر','فون','PS','حمایت','ووٹ','نوٹ'];
  sheet.appendRow(headers);
  data.voters.forEach(function(v){
    sheet.appendRow([v.name,v.dad,v.cnic,v.house,v.block,v.gender,v.age,v.phone,v.ps,v.status,v.voted?'ہاں':'نہیں',v.note]);
  });
  return ContentService.createTextOutput(JSON.stringify({ok:true,total:data.voters.length})).setMimeType(ContentService.MimeType.JSON);
}
